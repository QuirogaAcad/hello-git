git             # Ejecuta Git y muestra ayuda básica o lista de comandos disponibles
git --version   # Muestra la versión de Git instalada en la computadora
clear           # Limpia la pantalla de la terminal
ls              # Lista los archivos y carpetas del directorio actual
cd              # Cambia de directorio (navegar entre carpetas)
pwd             # Muestra la ruta completa de la carpeta actual
mkdir           # Crea una nueva carpeta (directorio)
code .          # Abre la carpeta actual en Visual Studio Code
git config --global user.name "Gab0"                    # Configura tu nombre de usuario global para los commits
git config --global user.email "quiroga.acad@gmail.com" # Configura tu email global para identificar tus commits
touch hellogit.py    # Crea un archivo vacío llamado hellogit.py
git init             # Inicializa un repositorio Git en la carpeta actual (empieza el control de versiones)
git branch -m main   # Cambia el nombre de la rama actual a "main"
git status           # Muestra el estado del repositorio (archivos nuevos, modificados o listos para commit)
git add hellogit.py  # Agrega ese archivo al área de preparación (staging) para el próximo commit
git add .            # Agrega TODOS los archivos modificados al área de preparación
git commit -m "Este es mi primer commit"  # Guarda un punto de versión del proyecto con un mensaje descriptivo
git log              # Muestra el historial completo de commits del repositorio
git checkout         # Cambia de rama o restaura archivos a versiones anteriores
git reset            # Quita archivos del área de preparación o mueve el repositorio a un commit anterior
git log --graph      # Muestra el historial de commits con un gráfico de ramas en texto
git log --graph --prety=oneline             # Muestra el historial en una sola línea por commit (el correcto es --pretty)
git log --graph --decorate --all --oneline  # Muestra TODAS las ramas con gráfico, nombres de ramas y commits resumidos
git config --global alias.arbol "log --graph --decorate --all --oneline"  # Crea un alias llamado "arbol" para ese comando largo
git arbol            # Ejecuta el alias creado y muestra el árbol visual de commits
touch .gitignore     # Crea el archivo .gitignore para indicar qué archivos Git debe ignorar
git diff             # Muestra las diferencias entre los cambios actuales y el último commit
git checkout HEAD    # Restaura los archivos al estado del último commit (descarta cambios no guardados)
git reset --hard     # Vuelve el repositorio exactamente al último commit y elimina TODOS los cambios locales
git reflog           # Muestra el historial interno de movimientos de HEAD (sirve para recuperar commits perdidos)
git tag              # Lista o crea etiquetas para marcar versiones del proyecto
git checkout tag/dia1  # Cambia el repositorio al estado guardado en el tag llamado "dia1"
git branch login     # Crea una nueva rama llamada "login"
git switch login     # Cambia a la rama llamada "login"
git merge main       # Fusiona los cambios de la rama "main" en la rama actual
git stash            # Guarda temporalmente los cambios no confirmados (sin commit) para limpiar el área de trabajo
git stash list       # Muestra la lista de cambios guardados en el stash
git stash pop        # Recupera el último cambio guardado en el stash y lo elimina de la lista
git stash drop       # Elimina un stash específico sin aplicarlo
git diff login       # Muestra las diferencias entre la rama actual y la rama llamada "login"
git branch -d login  # Elimina la rama llamada "login" (solo si ya fue fusionada)
git remote add origin git@github.com:QuirogaAcad/hello-git.git
git push -u origin main
